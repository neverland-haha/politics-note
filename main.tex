	\dimen@ = \dimexpr\wd\@tempboxa-\labelwidth\relax
	\labelwidth = \wd\@tempboxa
	\advance\linewidth -\dimen@
	\advance\leftmargin \dimen@
	\advance\@totalleftmargin \dimen@
	\parshape \@ne \@totalleftmargin \linewidth}
